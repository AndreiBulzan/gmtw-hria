=== GMTW-Ro Commands ===

=== 1. GENERATE DATASETS ===

python scripts/generate_gmtw_v0.py --num-travel 150 --num-schedule 150 --num-fact 100 --num-recipe 100 --output data/gmtw_ro_v0.jsonl

python scripts/generate_gmtw_hard.py --num-travel 100 --num-schedule 75 --num-fact 50 --num-recipe 75 --output data/gmtw_ro_hard.jsonl


=== 2. RUN MODELS ON BASE DATASET (500) ===

python scripts/run_groq_batch.py data/gmtw_ro_v0.jsonl --model openai/gpt-oss-20b --output data/outputs_base_gpt-oss-20b.jsonl --delay 2.0

python scripts/run_groq_batch.py data/gmtw_ro_v0.jsonl --model meta-llama/llama-4-scout-17b-16e-instruct --output data/outputs_base_llama4-scout.jsonl --delay 2.0

python scripts/run_groq_batch.py data/gmtw_ro_v0.jsonl --model meta-llama/llama-guard-4-12b --output data/outputs_base_llama-guard.jsonl --delay 2.0

python scripts/run_groq_batch.py data/gmtw_ro_v0.jsonl --model qwen/qwen3-32b --output data/outputs_base_qwen3-32b.jsonl --delay 2.0

python scripts/run_groq_batch.py data/gmtw_ro_v0.jsonl --model llama-3.3-70b-versatile --output data/outputs_base_llama33-70b.jsonl --delay 2.0


=== 3. RUN MODELS ON HARD DATASET (300) ===

python scripts/run_groq_batch.py data/gmtw_ro_hard.jsonl --model openai/gpt-oss-20b --output data/outputs_hard_gpt-oss-20b.jsonl --delay 2.0

python scripts/run_groq_batch.py data/gmtw_ro_hard.jsonl --model meta-llama/llama-4-scout-17b-16e-instruct --output data/outputs_hard_llama4-scout.jsonl --delay 2.0

python scripts/run_groq_batch.py data/gmtw_ro_hard.jsonl --model llama-3.3-70b-versatile --output data/outputs_hard_llama33-70b.jsonl --delay 2.0


=== 4. EVALUATE BASE DATASET RESULTS ===

python scripts/evaluate_outputs.py data/gmtw_ro_v0.jsonl data/outputs_base_gpt-oss-20b.jsonl --save-metrics data/metrics_base_gpt-oss-20b.jsonl

python scripts/evaluate_outputs.py data/gmtw_ro_v0.jsonl data/outputs_base_llama4-scout.jsonl --save-metrics data/metrics_base_llama4-scout.jsonl
